-- TABLE Movie
CREATE TABLE Movie (
MovieID varchar(10) NOT NULL PRIMARY KEY,
MovieTitle varchar (30) NOT NULL,
Popularity float NOT NULL,
ImdbScore float NOT NULL,
ReleaseDate DATE NOT NULL,
fk_Genre integer REFERENCES Genre(GenreID),
fk_Director integer REFERENCES Director(DirectorID)
);

-- TABLE Genre;
CREATE TABLE Genre(
GenreID varchar(10) NOT NULL PRIMARY KEY,
GenreType varchar(30) NOT NULL
);

-- TABLE Director;
CREATE TABLE Director(
DirectorID varchar(10) NOT NULL PRIMARY KEY,
DirectorName varchar(30) NOT NULL,
DirectoryNationality varchar(30) NOT NULL
);

-- TABLE User;
CREATE TABLE USer(
id = fields.IntField(pk=True)
name = fields.CharField(250)
password = fields.CharField(100)
username = fields.CharField(100, unique=True)
is_admin = fields.BooleanField(default=False)
is_active = fields.BooleanField(default=True)
);

-- TABLE Token;
CREATE TABLE Token(
fk_user integer REFERENCES Users(UserID),
token varchar (100) NOT NULL,
);
